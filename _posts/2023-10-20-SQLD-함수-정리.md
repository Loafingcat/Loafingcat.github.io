[SUM]

col1=1, col2=null

SUM(col1+col2)=null

SUM(col1)+SUM(col2)=1

[COALESCE]

COALESCE(NULL, '2')=2

COALESCE('2', NULL)=2 

하나 이상의 매개변수를 가질 수 있음.

[NULLIF]

NULLIF('A', 'A')=NULL

NULLIF('B', 'C')=B

매개변수가 동일하지 않을 경우 첫 번째 매개변수를 반환함.

[NVL]

두 개의 인수가 필요함.

NVL(NULL, 0)=0

NVL(NULL, 2)=2

[NVL2]

NVL2(값, A, B)

NVL2(NOT NULL, A, B)=A

NVL2(NULL, A, B)=B
